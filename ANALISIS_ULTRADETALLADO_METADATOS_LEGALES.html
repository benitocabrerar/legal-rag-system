// Validar resultados web contra metadatos locales async function crossReferenceResults( localResults: ChunkWithMetadata[], webResults: WebSearchResult[], options: ValidationOptions ): Promise { return webResults.map(webResult => { const validations = []; // 1. Verificar número de Registro Oficial if (options.verifyPublicationNumbers) { const roMatch = webResult.content.match(/RO\s+(\d+)/); if (roMatch) { const localHasRO = localResults.some( r => r.metadata.publicationNumber === roMatch[1] ); validations.push({ type: 'publication_number', valid: localHasRO, confidence: localHasRO ? 0.95 : 0.6 }); } } // 2. Verificar fechas de publicación if (options.checkDates) { const dateMatch = extractDate(webResult.content); const dateValid = localResults.some(r => isSimilarDate(r.metadata.publicationDate, dateMatch) ); validations.push({ type: 'date', valid: dateValid, confidence: dateValid ? 0.85 : 0.5 }); } // 3. Verificar jurisdicción if (options.matchJurisdiction) { const jurisdictionMatch = detectJurisdiction(webResult.content); validations.push({ type: 'jurisdiction', valid: jurisdictionMatch, confidence: 0.75 }); } // Score final de confianza const avgConfidence = validations.reduce( (sum, v) => sum + v.confidence, 0 ) / validations.length; return { ...webResult, validations, trustScore: avgConfidence, isVerified: avgConfidence > 0.7 }; }); }
